# 02. Sorted Reads

Los BAMs de cada corrida se ordenan por coordenada genómica con `samtools sort`.

- **Input:** BAMs por corrida en `/data2/lab2/lucila/BIO295B_FernandaCavieres/data/bam_completo_norte_final/`
- **Output:** `*_sorted.bam` en el mismo directorio
- **Herramienta:** `samtools sort` (conda env: `preSNPcalling`), 8 threads

---

## *S. parasiticus* — *script 02a.sort_sparasiticus.sh*
#!/bin/bash
export TZ="America/Santiago"
source activate preSNPcalling

bam="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/bam_completo_norte_final"

files="SPA1 SPA9 \
       SRR22267306 SRR10019934 SRR10019945 SRR9946748 \
       JJW5680_high1 JJW5680_high2 JJW5680_low \
       JJW4274_high1 JJW4274_high2 JJW4274_low \
       JJW4273_high1 JJW4273_high2 JJW4273_low \
       JJW730_high1 JJW730_high2 JJW730_low \
       T2-15_high1 T2-15_high2 T2-15_low \
       T7-11_high1 T7-11_high2 T7-11_low \
       HK_2019_high1 HK_2019_high2 HK_2019_low \
       UAMX5074_high1 UAMX5074_high2 UAMX5074_low \
       UAMX602_high1 UAMX602_high2 UAMX602_low \
       UAMX623_high1 UAMX623_high2 UAMX623_low \
       ZR_2019_high1 ZR_2019_high2 ZR_2019_low \
       AM_2019_high1 AM_2019_high2 AM_2019_low \
       AX_2016_high AX_2016_low \
       JX_2019_high JX_2019_low \
       T1-15_high T1-15_low \
       T14-19_high T14-19_low \
       T3-10_high T3-10_low \
       T4-10_high T4-10_low \
       TB1-15_high TB1-15_low \
       TB1-16_high TB1-16_low \
       TB2-16_high TB2-16_low \
       TB3-11_high TB3-11_low \
       TB4-11_high TB4-11_low \
       TB6-11_high TB6-11_low \
       TS17-16_high TS17-16_low"

for sample in $files; do
    echo "[$(date '+%H:%M:%S')] Sorting: $sample"
    samtools sort -@ 8 \
        -o $bam/${sample}_sorted.bam \
        $bam/${sample}.bam && \
    echo "[$(date '+%H:%M:%S')] OK: $sample" || \
    echo "[$(date '+%H:%M:%S')] ERROR: $sample"
done

echo "============================================"
echo "Sort S. parasiticus terminado"
echo "============================================"
