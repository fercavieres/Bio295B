# 09. MC1R whatshap phase — *S. longicaudus*

```bash
#!/bin/bash
export TZ="America/Santiago"
source activate whatshap-env

REF="/data2/lab2/lucila/01.anotacion/SterChil1.hic.hap1.yahs_scaffolds_final.fa.MicroFinder.ordered.fa.masked"
BAMDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/bam_completo_norte_final"
VCFDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_longicaudus/vcf"
PHASEDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_longicaudus/phased"

mkdir -p "$PHASEDIR"

MUESTRAS=(
    "SL5" "SL8" "SRR22267307"
    "UAMX160" "UAMX161"
    "JJW1966" "JJW1467" "JJW5548"
    "TH-19" "OJ-19"
    "F1-21" "F2-21" "F3-21" "F4-21" "F5-21" "F6-21"
    "KT-19" "OK-19"
)

echo "============================================"
echo " PASO 2: whatshap phase"
echo " Inicio: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"

for NOMBRE in "${MUESTRAS[@]}"; do
    VCF="$VCFDIR/${NOMBRE}.vcf.gz"
    BAM="$BAMDIR/${NOMBRE}.realign.bam"
    if [ -f "$VCF" ] && [ -f "$BAM" ]; then
        echo "Faseando $NOMBRE..."
        whatshap phase \
            --reference "$REF" \
            --output "$PHASEDIR/${NOMBRE}_phased.vcf.gz" \
            "$VCF" "$BAM"
        bcftools index "$PHASEDIR/${NOMBRE}_phased.vcf.gz"
        echo "✓ $NOMBRE faseado"
    else
        echo "✗ Falta VCF o BAM para $NOMBRE"
    fi
done

echo "============================================"
echo " PASO 2 terminado: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"
```
