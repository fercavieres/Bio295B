# 03. MC1R bcftools mpileup + call — *S. pomarinus*

```bash
#!/bin/bash
export TZ="America/Santiago"
source activate mc1r_env

REF="/data2/lab2/lucila/01.anotacion/SterChil1.hic.hap1.yahs_scaffolds_final.fa.MicroFinder.ordered.fa.masked"
BAMDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/bam_completo_norte_final"
VCFDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_pomarinus/vcf"
REGION="scaffold_12:41127919-41128863"

mkdir -p "$VCFDIR"

MUESTRAS=(
    "SPO4" "SPO7"
    "SRR22267299" "SRR22267300" "SRR22267301"
    "ABJ189" "JJW2216" "UAMX383" "UAMX8584"
)

echo "============================================"
echo " PASO 1: bcftools mpileup + call"
echo " Inicio: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"

for NOMBRE in "${MUESTRAS[@]}"; do
    BAM="$BAMDIR/${NOMBRE}.realign.bam"
    if [ -f "$BAM" ]; then
        echo "Procesando $NOMBRE..."
        bcftools mpileup -r "$REGION" -f "$REF" -q 20 -Q 20 "$BAM" \
            | bcftools call -mv -Oz -o "$VCFDIR/${NOMBRE}.vcf.gz"
        bcftools index "$VCFDIR/${NOMBRE}.vcf.gz"
        echo "✓ $NOMBRE VCF listo"
    else
        echo "✗ No se encontró BAM: $BAM"
    fi
done

echo "============================================"
echo " PASO 1 terminado: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"
```
