# 01. MC1R Haplotype Extraction — *S. parasiticus*

Extracción de haplotipos del gen MC1R (`scaffold_12:41127919-41128863`, 945 bp, exón único) para las 31 muestras de *S. parasiticus*. El pipeline corre en tres pasos secuenciales.

- **Input:** `*.realign.bam` por muestra
- **Output final:** `MC1R_parasiticus_haplotipos.fasta` (63 secuencias: 1 referencia + 31×2 haplotipos)
- **Directorios:**
  - VCFs: `data/mc1r_parasiticus/vcf/`
  - VCFs faseados: `data/mc1r_parasiticus/phased/`
  - Haplotipos: `data/mc1r_parasiticus/haplotipos/`

---

## Paso 1 — bcftools mpileup + call — `01a.mpileup_parasiticus.sh`

```bash
#!/bin/bash
export TZ="America/Santiago"
source activate mc1r_env

REF="/data2/lab2/lucila/01.anotacion/SterChil1.hic.hap1.yahs_scaffolds_final.fa.MicroFinder.ordered.fa.masked"
BAMDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/bam_completo_norte_final"
VCFDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_parasiticus/vcf"
REGION="scaffold_12:41127919-41128863"

mkdir -p "$VCFDIR"

MUESTRAS=(
    "JJW5680" "JJW4274" "JJW4273" "JJW730"
    "T1-15" "T14-19" "T2-15" "T3-10" "T4-10" "T7-11"
    "TB1-15" "TB1-16" "TB2-16" "TB3-11" "TB4-11" "TB6-11" "TS17-16"
    "AM_2019" "AX_2016" "JX_2019" "ZR_2019" "HK_2019"
    "UAMX5074" "UAMX602" "UAMX623"
    "SPA9" "SPA1"
    "SRR22267306" "SRR10019934" "SRR10019945" "SRR9946748"
)

echo "============================================"
echo " PASO 1: bcftools mpileup + call"
echo " Inicio: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"

for NOMBRE in "${MUESTRAS[@]}"; do
    BAM="$BAMDIR/${NOMBRE}.realign.bam"
    if [ -f "$BAM" ]; then
        echo "Procesando $NOMBRE..."
        bcftools mpileup -r "$REGION" -f "$REF" -q 20 -Q 20 "$BAM" \
            | bcftools call -mv -Oz -o "$VCFDIR/${NOMBRE}.vcf.gz"
        bcftools index "$VCFDIR/${NOMBRE}.vcf.gz"
        echo "✓ $NOMBRE VCF listo"
    else
        echo "✗ No se encontró BAM: $BAM"
    fi
done

echo "============================================"
echo " PASO 1 terminado: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"
```

## Paso 3 — bcftools consensus hap1 + hap2 — `01c.consensus_parasiticus.sh`

> El consensus se genera sobre el genoma completo y luego se extrae la región con `samtools faidx`. Esto evita el problema de que `bcftools consensus -r` generaba FASTAs de ~1.2G por muestra.

```bash
#!/bin/bash
export TZ="America/Santiago"
source activate mc1r_env

REF="/data2/lab2/lucila/01.anotacion/SterChil1.hic.hap1.yahs_scaffolds_final.fa.MicroFinder.ordered.fa.masked"
SAMTOOLS="/home/lab2/.conda/envs/mc1r_env/bin/samtools"
PHASEDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_parasiticus/phased"
HAPDIR="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_parasiticus/haplotipos"
REGION="scaffold_12:41127919-41128863"

mkdir -p "$HAPDIR"

MUESTRAS=(
    "JJW5680" "JJW4274" "JJW4273" "JJW730"
    "T1-15" "T14-19" "T2-15" "T3-10" "T4-10" "T7-11"
    "TB1-15" "TB1-16" "TB2-16" "TB3-11" "TB4-11" "TB6-11" "TS17-16"
    "AM_2019" "AX_2016" "JX_2019" "ZR_2019" "HK_2019"
    "UAMX5074" "UAMX602" "UAMX623"
    "SPA9" "SPA1"
    "SRR22267306" "SRR10019934" "SRR10019945" "SRR9946748"
)

echo "============================================"
echo " PASO 3: bcftools consensus hap1 + hap2"
echo " Inicio: $(date '+%Y-%m-%d %H:%M:%S')"
echo "============================================"

for NOMBRE in "${MUESTRAS[@]}"; do
    PHASED="$PHASEDIR/${NOMBRE}_phased.vcf.gz"
    if [ -f "$PHASED" ]; then
        echo "Procesando $NOMBRE..."

        # Haplotipo 1
        bcftools consensus --haplotype 1 -f "$REF" "$PHASED" \
            -o "$HAPDIR/${NOMBRE}_hap1_full.fa"
        $SAMTOOLS faidx "$HAPDIR/${NOMBRE}_hap1_full.fa"
        $SAMTOOLS faidx "$HAPDIR/${NOMBRE}_hap1_full.fa" "$REGION" \
            | sed "s/^>.*/>$NOMBRE\_hap1/" > "$HAPDIR/${NOMBRE}_hap1.fa"
        rm "$HAPDIR/${NOMBRE}_hap1_full.fa" "$HAPDIR/${NOMBRE}_hap1_full.fa.fai"

        # Haplotipo 2
        bcftools consensus --haplotype 2 -f "$REF" "$PHASED" \
            -o "$HAPDIR/${NOMBRE}_hap2_full.fa"
        $SAMTOOLS faidx "$HAPDIR/${NOMBRE}_hap2_full.fa"
        $SAMTOOLS faidx "$HAPDIR/${NOMBRE}_hap2_full.fa" "$REGION" \
            | sed "s/^>.*/>$NOMBRE\_hap2/" > "$HAPDIR/${NOMBRE}_hap2.fa"
        rm "$HAPDIR/${NOMBRE}_hap2_full.fa" "$HAPDIR/${NOMBRE}_hap2_full.fa.fai"

        echo "✓ $NOMBRE listo"
    else
        echo "✗ No se encontró VCF faseado para $NOMBRE"
    fi
done

# Juntar todos los haplotipos
echo ""
echo "Juntando haplotipos..."
samtools faidx "$REF" "$REGION" | sed "s/^>.*/>SC_chilensis_ref/" > "$HAPDIR/MC1R_parasiticus_haplotipos.fasta"
for NOMBRE in "${MUESTRAS[@]}"; do
    cat "$HAPDIR/${NOMBRE}_hap1.fa" >> "$HAPDIR/MC1R_parasiticus_haplotipos.fasta"
    cat "$HAPDIR/${NOMBRE}_hap2.fa" >> "$HAPDIR/MC1R_parasiticus_haplotipos.fasta"
done

echo "============================================"
echo " LISTO! $(date '+%Y-%m-%d %H:%M:%S')"
echo " Archivo final: $HAPDIR/MC1R_parasiticus_haplotipos.fasta"
grep -c "^>" "$HAPDIR/MC1R_parasiticus_haplotipos.fasta"
echo " secuencias en total (ref + 2 hap x muestra)"
echo "============================================"
```

## Paso 4 — Traducción a proteína — `01d.translate_parasiticus.sh`

```bash
#!/bin/bash
export TZ="America/Santiago"
source activate jcvi_env

DIR_IN="/data2/lab2/lucila/BIO295B_FernandaCavieres/data/mc1r_parasiticus/haplotipos"
DIR_OUT="$DIR_IN/proteinas"
mkdir -p "$DIR_OUT"

echo "Traduciendo haplotipos MC1R a proteinas..."
seqkit translate -F "$DIR_IN/MC1R_parasiticus_haplotipos.fasta" > "$DIR_OUT/MC1R_parasiticus_proteinas.faa"

echo "Traduccion completa!"

echo "Revisando codones STOP internos..."
if grep -v "\*$" "$DIR_OUT/MC1R_parasiticus_proteinas.faa" | grep -q "\*"; then
    echo "⚠️ Hay STOPs internos (revisar secuencias)"
else
    echo "✓ Solo STOP final (todo correcto)"
fi

echo "Resumen de longitudes:"
seqkit stats "$DIR_OUT/MC1R_parasiticus_proteinas.faa"
```
